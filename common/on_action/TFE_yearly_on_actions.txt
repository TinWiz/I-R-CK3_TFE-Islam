# IRToCK3: Last updated for CK3 1.7.2

# Called every Jan 1st
# Root does not exist
yearly_global_pulse = {
	on_actions = { TFE_yearly_global_pulse }
}

TFE_yearly_global_pulse = {
	effect = {
		#Spawning islam
		if = {
			limit = {
				current_date >= 622.1.1
				NOT = { exists = global_var:islam_spawned }
			}
			trigger_event = {
				id = islam.0001
				days = {30 50}
			}
			set_global_variable = {
				name = islam_spawned
				value = yes
			}
		}
	}
}
